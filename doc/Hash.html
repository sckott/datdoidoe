def links(just_urls = true)
  if self['message']['items'].nil?
    tmp = self['message']['link']
    if tmp.nil?
      tmp = nil
    else
      tmp = tmp.reject { |c| c.empty? }
    end
  else
    tmp = self['message']['items'].collect { |x| x['link'] }.reject { |c| c.empty? }
  end

	return parse_links(tmp, just_urls)
end
