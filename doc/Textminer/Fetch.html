def fetchtext
  lks = Textminer.links(self.doi)
  lk = pick_link(lks)
  case self.type
  when "xml"
    # HTTParty.get(lk)
    coll = []
    Array(lk).each do |x|
      coll << HTTParty.get(x)
    end
    return coll
  when "pdf"
    serialize_pdf(lk, self.doi)
  end
end
