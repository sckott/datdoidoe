def to_s
  if !@doi.nil?
    if @doi.length > 3
      ending = '...'
    else
      ending = ''
    end
    tt = sprintf('dois: %s %s', Array(@doi)[0..2].join(', '), ending)
  end
  if !@member.nil?
    tt = 'member: ' + @member.to_s
  end
  if @doi.nil? && @member.nil?
    tt = ''
  end
  sprintf("<textminer>: \n      search: %s\n      no. licenses: %s", tt, @facet)
end
